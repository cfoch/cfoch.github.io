 * The prime factors of 13195 are 5, 7, 13 and 29.
 * What is the largest prime factor of the number 600851475143 ?
*/

#include <stdio.h>
#include <glib.h>

gboolean primo (gint64 numero) {
	
	gint64 j;
	gboolean is_primo;
	
	if (numero != 2) {
		is_primo = TRUE;
		j = 2;
		while ((j < numero) && (is_primo)) {
			if (numero % j == 0) {
				is_primo = TRUE;
			}
			j++;
		}
	}
	else {
		is_primo = TRUE;
	}
	return is_primo;	
	
}
